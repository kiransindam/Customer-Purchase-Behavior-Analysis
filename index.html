Core DAX MeasuresPower BI
// Paste into: Modeling → New Measure Total Revenue = SUM( CustomerTrends[purchase_amount] ) Avg Purchase = AVERAGE( CustomerTrends[purchase_amount] ) Customer Count = DISTINCTCOUNT( CustomerTrends[customer_id] ) Avg Rating = AVERAGE( CustomerTrends[review_rating] ) % Subscribed = DIVIDE( CALCULATE( [Customer Count], CustomerTrends[subscription_status] = "Yes" ), [Customer Count] ) Revenue % of Total = DIVIDE( [Total Revenue], CALCULATE( [Total Revenue], ALL( CustomerTrends ) ) ) Category Revenue Rank = RANKX( ALL( CustomerTrends[category] ), [Total Revenue] )
Python Analysis Scriptanalysis.py
import pandas as pd import matplotlib.pyplot as plt import seaborn as sns import os # Setup sns.set_theme(style='whitegrid', palette='muted') os.makedirs('charts', exist_ok=True) # Load & clean df = pd.read_csv('../data/customer_trends_analysis.csv') df['age_group'] = df['age_group'].replace( 'Aduit', 'Adult') # Revenue by category cat = df.groupby('category')['purchase_amount']\ .sum().sort_values(ascending=False) fig, ax = plt.subplots(figsize=(9, 5)) sns.barplot(x=cat.index, y=cat.values, palette='Blues_d', ax=ax) ax.set_title('Revenue by Category', fontsize=14, fontweight='bold') fig.tight_layout() fig.savefig('charts/01_revenue_by_category.png', dpi=150) # Seasonal box plot fig, ax = plt.subplots(figsize=(10, 6)) sns.boxplot(data=df, x='season', y='purchase_amount', order=['Spring','Summer','Fall','Winter'], palette='Set2', ax=ax) fig.savefig('charts/04_seasonal_sales.png', dpi=150) # Correlation heatmap nums = ['age','purchase_amount', 'review_rating','previous_purchases'] fig, ax = plt.subplots(figsize=(7, 5)) sns.heatmap(df[nums].corr(), annot=True, cmap='coolwarm', fmt='.2f', ax=ax) fig.savefig('charts/07_correlation_heatmap.png', dpi=150) print('✅ All charts saved to /charts/')
